PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX oh: <http://www.semanticweb.org/annag/ontologies/2024/1/ontoHorses#>

select ?cavalliInVendita ?Regione ?Cavaliere
where {
    ?cavalliInVendita rdf:type oh:Cavallo;
    		oh:èAdattoPer ?Cavaliere;
    		oh:regioneDiScuderizzazione ?Regione.

    FILTER (STR(?Regione)=STR("Piemonte") && ?Cavaliere=oh:CavalierePrincipiante )
   
    
} 


PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX schema: <http://schema.org/image>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX oh: <http://www.semanticweb.org/annag/ontologies/2024/1/ontoHorses#>

select distinct ?Razza ?Nazionalità ?Immagine 
where {
    	?Razza rdf:type oh:RazzaCavallo;
    			oh:haNazionalità ?Nazionalità.
    OPTIONAL{
    	?entita owl:sameAs ?Razza
    			SERVICE <https://query.wikidata.org/sparql> {
					OPTIONAL {
						?entita <http://www.wikidata.org/prop/direct/P18> ?Immagine.
 						}
 					
					 }
    }
    
    FILTER(STRSTARTS(STR(?Razza), STR(oh:))).	     
}
ORDER BY ?Nazionalità


PREFIX oh: <http://www.semanticweb.org/annag/ontologies/2024/1/ontoHorses#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

select ?Razza ?Morfologia ?Indole ?Regione ?Nazionalità ?AltezzaGarrese ?PesoMedio ?Impiego ?ColoriMantello
where {
       ?Razza rdf:type oh:RazzaCavallo;
    			oh:haIndole ?Indole;
    			oh:haRegioneDiOrigine ?Regione;
    			oh:haNazionalità ?Nazionalità;
    			oh:altezzaMediaAlGarrese_cm ?AltezzaGarrese;
    			oh:pesoMedio_Kg ?PesoMedio;
    			oh:puòEssereImpiegataIn ?Impiego;
    			oh:haColoreMantello ?ColoriMantello.
 FILTER(STRSTARTS(STR(?Razza), STR(oh:)) && ?Razza=oh:Murgese). 	
}
